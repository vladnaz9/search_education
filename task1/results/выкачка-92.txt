    var imgs = document.querySelectorAll("#photo_gallery img");

    for (var i = 0; i < imgs.length; i++) {
        var img = imgs[i];

        function imgLoad(){
            // document.querySelector("#test").innerHTML += ' ' + this.naturalWidth;
            this.aspect = this.naturalWidth / this.naturalHeight;
            photoGalleryUpdate();
        }

        function imgLoad2(image){
            // document.querySelector("#test").innerHTML += ' ' + this.naturalWidth;
            image.aspect = image.naturalWidth / image.naturalHeight;
            photoGalleryUpdate();
        }

        img.addEventListener('load', imgLoad, false);
        setTimeout(imgLoad2,300,img);

    }

    function photoGalleryUpdate(){
        var imgsCont = document.querySelector("#photo_gallery");
        var imgs = document.querySelectorAll("#photo_gallery img");
        var tiles = document.querySelectorAll("#photo_gallery .photo_gallery_tile");

        var imgsContRect = imgsCont.getBoundingClientRect();


        var imgcount = 0;                    
        var rowWidth = 0;                    
        var rowHeight = 0;                    
        var rowHeightAll = 0;                    
        for (var i = 0; i < imgs.length; i++) {
            tiles[i].style.zIndex = imgs.length - i;
            tiles[i].style.opacity = 0;
            
            if(typeof imgs[i].aspect !== "undefined") {

                imgcount++;
                rowWidth += imgs[i].naturalWidth;
                if(rowWidth > imgsContRect.width) {
                    var rowAspect = (imgsContRect.width+1) / rowWidth;
                    // console.log(rowAspect,imgcount);
                    var imgH = 150 * rowAspect;
                    rowHeightAll += imgH;
                    
                    rowWidth = 0;

                    for (var j = imgcount-1; j > -1; j--) {
                        var imgW = imgH * imgs[i-j].aspect;

                        // if(rowWidth + imgW > imgsContRect.width) imgW = imgsContRect.width - rowWidth;

                        tiles[i-j].style.height = imgH - 1 + "px";
                        tiles[i-j].style.width = imgW - 1 + "px";

                        tiles[i-j].style.top = rowHeightAll - imgH + "px";
                        tiles[i-j].style.left = rowWidth + "px";
                        
                        tiles[i-j].style.opacity = 1;            

                        rowWidth += imgW;
                        // console.log(imgs[i-j].style.height);
                    }
                    
                    rowWidth = 0;
                    imgcount = 0;
                }
            }
        }

        imgsCont.style.height = rowHeightAll + "px"

    }

    window.photoGalleryUpdateTimeout = null;

    window.addEventListener("resize",function(){
        clearTimeout(photoGalleryUpdateTimeout);
        photoGalleryUpdateTimeout = setTimeout(function(){
            photoGalleryUpdate();
        },100);
    })

</script>
